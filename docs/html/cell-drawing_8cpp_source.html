108 /// @brief Upload all Cells in the Universe to the CellInstancing.buffer so that they can be drawn.
109 ///
110 /// Overwrites the buffer each call, so any updates are drawn. CellInastances could be updated
111 /// individually more cleverly, but it isn't worth the complexity while re-uploading them each
112 /// frame is fast enough.
113 ///
114 void
115 upload_cell_instances(Universe *universe, CellInstancing *cell_instancing)
116 {
117  // Zero the buffer
118  cell_instancing->buffer.elements_used = 0;
119 
120  u32 i = 0;
121  for (u32 hash_slot = 0;
122  hash_slot < universe->hashmap_size;
123  ++hash_slot)
124  {
125  CellBlock *cell_block = universe->hashmap[hash_slot];
126 
127  if (cell_block != 0 && cell_block->initialised)
128  {
129  do
130  {
131  for (u32 cell_y = 0;
132  cell_y < CELL_BLOCK_DIM;
133  ++cell_y)
134  {
135  for (u32 cell_x = 0;
136  cell_x < CELL_BLOCK_DIM;
137  ++cell_x)
138  {
139  Cell *cell = cell_block->cells + (cell_y * CELL_BLOCK_DIM) + cell_x;
140 
141  vec2 cell_position = vec2_divide(uvec2_to_vec2(cell->block_offset), CELL_BLOCK_DIM);
142 
143  vec4 colour;
144  if (cell->state == 1)
145  {
146  colour = (vec4){0.3, 0.3, 0.3, 1};
147  }
148  else
149  {
150  colour = (vec4){0.7, 0.7, 0.7, 1};
151  }
152 
153  CellInstance cell_instance = {
154  .block_position = cell_block->block_position,
155  .cell_position = cell_position,
156  .colour = colour
157  };
158 
159  opengl_buffer_new_element(&cell_instancing->buffer, &cell_instance);
160  }
161  }
162 
163  // Follow any hashmap collision chains
164  cell_block = cell_block->next_block;
165  }
166  while (cell_block != 0);
167  }
168  }
169 }
